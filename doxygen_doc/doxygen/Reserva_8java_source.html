86 public boolean haExcedidoTiempoLimite() {

88 LocalDateTime horaLimite = LocalDateTime.of(convertirDateALocalDate(fecha), hora);
89
90 // Calcula la duración entre la hora actual y la hora límite
91 Duration duracion = Duration.between(horaLimite, horaActual);
92 long minutosExcedidos = duracion.toMinutes();
93
94 // Comprueba si se ha excedido el tiempo límite

98 private static LocalDate convertirDateALocalDate(Date date) {
99 Calendar calendar = Calendar.getInstance();
100 calendar.setTime(date);
101 int year = calendar.get(Calendar.YEAR);
102 int month = calendar.get(Calendar.MONTH) + 1; // Los meses en Calendar están indexados desde 0
103 int day = calendar.get(Calendar.DAY_OF_MONTH);
104
105 return LocalDate.of(year, month, day);
106 }
