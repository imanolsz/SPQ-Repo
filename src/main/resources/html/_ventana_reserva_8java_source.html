268 public int parking(Date fecha){
269 int parkingLibre = 0;
270 parkingOcupado = 0;
271 List<ReservaData> reservas = Main.getExampleClient().getReservas();
272 for(ReservaData reserva: reservas){
273 if(reserva.getFecha().equals(fecha)){
274 parkingOcupado += reserva.getAparcamiento();
275 }
276 }
277 parkingLibre = parkingTotal-parkingOcupado;
278
279
280 return parkingLibre;
